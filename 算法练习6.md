
**https://leetcode.cn/problems/replace-all-s-to-avoid-consecutive-repeating-characters/**
```
class Solution {
public:
    string modifyString(string s) {
        int n=s.size();
        for(int i=0;i<n;i++)
        {
            if(s[i]=='?')
            {
                for(char ch='a';ch<='z';ch++)
                {
                    if((i==0||s[i-1]!=ch)&&(i==n-1||s[i+1]!=ch))
                    {
                        s[i]=ch;
                        break;
                    }
                }
            }
        }
        return s;
    }
};

**https://leetcode.cn/problems/teemo-attacking/**
```
class Solution {
public:
    int findPoisonedDuration(vector<int>& timeSeries, int duration) {
        int n=timeSeries.size();
        int ret=0;
        for(int i=0;i<n-1;i++)
        {
            int x=timeSeries[i+1]-timeSeries[i];
            if(x<duration)ret+=x;
            else ret+=duration;
        }
        return ret+duration;
    }
};

**https://leetcode.cn/problems/zigzag-conversion/**
```
class Solution {
public:
    string convert(string s, int numRows) {
        if(numRows==1)return s;
        string ret;
        int n=s.size();
        int k=2*numRows-2;
        for(int i=0;i<n;i+=k)
        {
            ret+=s[i];
        }
        for(int i=1;i<numRows-1;i++)
        {
          for(int j=i,t=k-i;j<n||t<n;j+=k,t+=k)
          {
            if(j<n)ret+=s[j];
            if(t<n)ret+=s[t];
          }
        }
        for(int i=numRows-1;i<n;i+=k)
        {
            ret+=s[i];
        }
        return ret;
    }
};

**https://leetcode.cn/problems/minimum-number-of-frogs-croaking/**
```
class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        int ret=0;
        int hash[200]={0};
        for(int i=0;i<croakOfFrogs.size();i++)
        {
            if(croakOfFrogs[i]=='c')
            {
                if(hash['k'])hash['k']--;
                else
                    ret++;
                hash['c']++;
            }
            if(croakOfFrogs[i]=='r')
            {
                if(hash['c'])hash['c']--;
                else return -1;
                hash['r']++;
            }
            if(croakOfFrogs[i]=='o')
            {
                if(hash['r'])hash['r']--;
                else return -1;
                hash['o']++;
            }
            if(croakOfFrogs[i]=='a')
            {
                if(hash['o'])hash['o']--;
                else return -1;
                hash['a']++;
            }
            if(croakOfFrogs[i]=='k')
            {
                if(hash['a'])hash['a']--;
                else return -1;
                hash['k']++;
            }
   
        }
        if(hash['c']||hash['r']||hash['o']||hash['a'])return -1;
        return ret;
    }
};
