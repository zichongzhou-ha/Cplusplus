//插入排序:时间复杂度(n*n)
void InsertOrder(int* arr, int n)
{
	for (int i = 0; i < n - 1; i++)
	{
		int end = i;
		int tmp = arr[end + 1];
		while (end >= 0)
		{
			if (arr[end] > tmp)
			{
				arr[end + 1] = arr[end];
				end--;
			}
			else
			{
				break;
			}
		}
		arr[end + 1] = tmp;
	}
}

void ShellOrder(int* arr, int n)
{
	int gap = n;
	while (gap > 1)
	{
		gap = gap / 3 + 1;
		for (int i = 0; i < n - gap; i++)
		{
			int end = i;
			int tmp = arr[end + gap];
			while (end >= 0)
			{
				if (arr[end] > tmp)
				{
					arr[end + gap] = arr[end];
					end-=gap;
				}
				else
				{
					break;
				}
			}
			arr[end + gap] = tmp;
		}
	}
	
}

void SelectOrder(int* arr, int n)
{
	int begin = 0;
	int end = n - 1;
	while (begin < end)
	{
		int mmax = end;
		int mmin = begin;
		for (int i = begin; i <= end; i++)
		{
			if (arr[mmin] > arr[i])mmin = i;
			if (arr[mmax] < arr[i])mmax = i;

		}
		Swap(&arr[begin], &arr[mmin]);
		if (mmax == begin)mmax = mmin;//很重要的一步，因为前面的交换
		Swap(&arr[end], &arr[mmax]);
		begin++;
		end--;
	}
}

void AdjustDown(int *arr,int root,int n)//递增就要建立大堆 这里的n是下标啊啊啊
{
	int parent = root;
	int child = 2 * root + 1;
	while (child<n)
	{
		if (child+1<n&&arr[child] < arr[child+1])child++;
		if (arr[child] > arr[parent])
		{
			Swap(&arr[child], &arr[parent]);
			parent = child;
			child = 2 * parent + 1;
		}
		else break;
	}
}
void HeapOrder(int* arr, int n)
{
	//先把堆建好
	for (int i = (n-1-1)/2; i >=0; i--)
	{
		AdjustDown(arr, i, n);
	}
	for (int i = 0; i < n - 1; i++)
	{
		Swap(&arr[0], &arr[n - 1-i]);
		AdjustDown(arr, 0, n - 1-i);
	}
	
}

int GetMidIndex(int* arr, int left, int right)
{
	int mid = left + (right - left) / 2;
	if (arr[right] < arr[left])
	{
		if (arr[mid] > arr[left])return left;
		else
		{
			if (arr[mid] < arr[right])return right;
			else return mid;
		}
	}
	else
	{
		if (arr[mid] > arr[right])return right;
		else
		{
			if (arr[mid] < arr[left])return left;
			else return mid;
		}
	}
}
//1.左右指针法
int PartOrder1(int *arr,int left,int right)
{
	int tmp = GetMidIndex(arr, left, right);//优化：不让最大或最小的数作为target
	Swap(&arr[tmp], &arr[right]);

	int target = right;
	while (left < right)
	{
		while (arr[left] <= arr[target] && left < right)//把目标值定在右边,那么在循环中就得先从左边开始
		{
			left++;
		}
		while (arr[right] > arr[target] && left < right)//如果都有=就导致有些在左有些在右错了;如果都不写就死循环
		{
			right--;
		}
		Swap(&arr[left], &arr[right]);
	}
	Swap(&arr[left], &arr[target]);
	return left;
}

int PartOrder2(int* arr, int left, int right)
{
	int tmp = GetMidIndex(arr, left, right);//优化：不让最大或最小的数作为target
	Swap(&arr[tmp], &arr[right]);

	int target = arr[right];
	while (left < right)
	{
		while (arr[left] < target && left < right)//把目标值定在右边,那么在循环中就得先从左边开始
		{
			left++;
		}
		arr[right] = arr[left];

		{
			right--;
		}
		arr[left] = arr[right];
	}
	arr[left] = target;
	return left;
}
//3. 前后指针法
int PartOrder3(int *arr,int left,int right)
{
	int tmp = GetMidIndex(arr, left, right);//优化：不让最大或最小的数作为target
	Swap(&arr[tmp], &arr[right]);
	int target = arr[right];
	int cur = left;
	int pre = left - 1;
	while (cur <= right)
	{
		if (arr[cur] <= target&&++pre!=cur)
		{
			Swap(&arr[cur],&arr[pre]);
		}
		cur++;
	}
	return pre;
}
void QuickOrder(int* arr, int left, int right)
{
	if (left >= right)return;
	int tmp=PartOrder3(arr, left, right);
	QuickOrder(arr, left, tmp - 1);
	QuickOrder(arr, tmp+1, right);
}

	if (right <= left)return;

//计数排序：耗空间
void CountSort(int* arr,int n)
{
	int max = arr[0];
	int min = arr[0];
	for (int i = 0; i < n; i++)
	{
		if (max < arr[i])max = arr[i];

		if (min > arr[i])min = arr[i];
	}
	int range = max - min + 1;
	int* p = (int*)malloc(sizeof(int) * range);
	memset(p, 0, sizeof(int) * range);
	for (int i = 0; i < n; i++)
	{
		p[arr[i] - min]++;
	}
	
	int index = 0;
	for (int i = 0; i < range;i++)
	{
		while (p[i]--)
		{
			arr[index++] = i + min;
		}
	}
}

**稳定性：数组中相同的值排完序后相对位置不变**

### 对排序的时间复杂度 空间复杂度 稳定性进行总结
- 直接插入排序
	1. 时间复杂度上，当数组已经有序了，那它是最快的，达到O(n),但是，一旦没有这个情况，就算是O(n^2);
	2. 稳定
- 希尔排序
	1. 希尔排序就是对插入排序的优化，在时间复杂度上平均为O(n^1.3);
	2. 不稳定
- 选择排序
	1. O(n^2)
	2. 稳定性上要特别注意，它是不稳定的，例如想把小的换过来，注意力都放在小的上了，以为把同样小的第一个小的交换过来就好了，但没想到被小的交换的那个数顺序也变了
- 堆排序
	1. O(n*logn)
	2. 不稳定
- 简单交换排序(冒泡排序)
	1. 当已经有序了，冒泡排序可以优化(设个flag)使得时间复杂度变为O(n),但一般情况下都是O(n^2);
	2. 稳定
- 快速排序
	1. 如果采用了优化，如三数取中，那么时间复杂度就是能达到O(n*logn),但如果没有，就为(0(n*logn)-O(n^2)),因为已经有序时反而变为n^2
	2. 不稳定，其实也有点难想：153465895--》153455896(这里就看出不稳了)--》153455698
- 归并排序
	1. nlog(n)
	2. 稳定
- 对这几种排序的空间复杂度进行分析一下
	- 堆排序其实并没有递归，一直在原数组操作，所以它没有额外的辅助空间
	- 快速排序递归了，所以肯定有栈帧的使用，辅助空间为O(logn)
	- 归并排序，有两处需要额外空间：递归(O(logn)),归并需要一个额外数组(O(n))，所以总体上是O(n);

### 外排序
- 例如，原题是有一堆1亿数据，内存只能存100w，要对这一亿数据进行排序，思路是什么
- 思路：首先，把这1亿数据分割为100个文件中，对每个文件里的数据进行排序(内存)，然后放回了硬盘，然后两两文件操作，打开文件，通过文件指针一个一个访问并两两比较，形成新的文件就写入到硬盘中，这个过程中不是说打开了那个文件就把那个文件全部加载到硬盘了，而是通过文件指针找到对应数据，所以内存每次都是读到两个数据，而不是读到两个文件里所有的数据。
- ai修正我的想法：硬盘读写的最小单位通常是一个块（如4KB），如果每次只读一条数据（比如8字节的整数），每读一条都要触发一次磁盘I/O，速度会慢得无法接受，实际外部排序中，我们会设置缓冲区（buffer），一次读取一批数据到内存（比如几千到几万条），然后进行归并
