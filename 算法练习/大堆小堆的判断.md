
### 判断一颗二叉树是大堆还是小堆，注意=
```
#include<iostream>
#include<vector>

using namespace std;
vector<int>nums;
int n;

static int dfs()
{
	bool isBig = true;
	bool isSmall = true;

	for (int i = n / 2; i >= 1; i--)
	{
		int left = 2 * i;
		int right = 2 * i + 1;
		if (left <= n && nums[left] > nums[i])isBig = false;
		if (left <= n && nums[left] < nums[i])isSmall = false;
		if (right <= n && nums[right] > nums[i])isBig = false;
		if (right <= n && nums[right]<nums[i])isSmall = false;
		if (!isBig && !isSmall)break;
	}
	if (!isBig && !isSmall)return 0;
	else if (isBig)return 1;
	else return 2;


}
int main()
{
	cin >> n;
	nums.resize(n + 1);
	for (int i = 1; i <= n; i++)
	{
		cin >> nums[i];
	}
	int ans = dfs();
	if (ans == 1)cout << "大堆" << endl;
	if (ans == 2)cout << "小堆" << endl;
	if (ans == 0)cout << "NONE" << endl;
	return 0;

}
```
