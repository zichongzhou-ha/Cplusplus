## 由后序遍历加上完全二叉树条件写出层序遍历
- 后序：左右中-----》操作---》得出中右左
- 这时候可以直接在纸上模拟出这个树形结构，结合完全二叉树的特点

```
#include<bits/stdc++.h>
using namespace std;
int ret[100],mp[100];
int n;

int solve(int a)
{
    //一定是先填的右边再填的左边，所以到着回去的时候一定是找左边的，不然有右的早就填了
    if(2*a+1<=n&&ret[2*a+1]==0)return 2*a+1;
    while(a>=1)
        {
            if(a*2<=n&&!ret[2*a])return 2*a;
            a/=2;
        }
    return a;
}
int main()
{
    cin>>n;
    for(int i=1;i<=n;i++)
        cin>>mp[i];
    int rt=1;
    for(int i=n;i>0;i--)
        {
            ret[rt]=mp[i];
            rt=solve(rt);
        }
    for(int i=1;i<=n;i++)
        {
            cout<<ret[i];
            if(i!=n)cout<<" ";
        }
    return 0;
}
```
