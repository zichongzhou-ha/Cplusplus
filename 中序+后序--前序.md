### 中序+后序----->前序

- 今天做题做的，竟然没写出来，忘了用哈希记录root在中序的位置


```

#include<iostream>
#include<vector>
#include<unordered_map>
using namespace std;
vector<int>mid_tree;
vector<int>post_tree;
pair<int,int>ans[1000];
unordered_map<int, int>p;
static int dfs(int midl, int midr, int postl, int postr)
{
	if (midl == midr)return post_tree[postr];
	if (midl > midr)return 0;
	int root = post_tree[postr];
	int pos = p[root];
	ans[root].first = dfs(midl,pos-1,postl,postl+(pos - midl)-1);
	ans[root].second = dfs(pos + 1, midr, postl + (pos - midl), postr - 1);
	return root;

}
static void Traverse(int root)
{
	if (root== 0)return;//叶子节点
	cout << root << " ";
	Traverse(ans[root].first);
	Traverse(ans[root].second);
}
int main()
{
	int n;
	cin >> n;
	mid_tree.resize(n);
	post_tree.resize(n);
	for (int i = 0; i < n; i++)cin >> mid_tree[i];
	for (int i = 0; i < n; i++)cin >> post_tree[i];
	for (int i = 0; i < n; i++)
	{
		p[mid_tree[i]] = i;
	}
	int root=dfs(0, n - 1, 0, n - 1);
	Traverse(root);
	return 0;
}
```
