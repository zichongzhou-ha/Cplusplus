## 智能指针

### 基础概念认知
- RAII是一种托管资源的思想，智能指针是依靠这种RAII实现的
```
#pragma once
template <class T>
class SmartPtr
{
public:
	SmartPtr(T* ptr)
		:_ptr(ptr)
	{}
	~SmartPtr()
	{
		if (_ptr)
		{
			cout << "delete:" << &_ptr;
			delete _ptr;
		}
	}
	T& operator*()
	{
		return *_ptr;

	}
	T* operator->()
	{
		return _ptr;
	}
private:
	T* _ptr;
};
```

### 解决智能指针的坑

- 问题：当多个智能指针指向的是同一份资源时，析构时就会导致同一个空间被析构多次
- 解决：
    - 管理权转移  C++98 auto_ptr
        - 缺陷：ap2=ap1场景下ap1就悬空了，访问就会报错
    - 防拷贝      C++11 unique_ptr
        - 简单粗暴，推荐使用，但如果要拷贝就不能用了
    - 引用计数的共享拷贝  C++11 shared_ptr


```
template <class T>
class auto_ptr
{
public:
	auto_ptr(T* ptr)
		:_ptr(ptr)
	{}
	auto_ptr(auto_ptr<T>& ap)
		:_ptr(ap._ptr)
	{
		ap._ptr = nullptr;
	}
	auto_ptr<T>& oprator=(const auto_ptr<T>& ap)
	{
		if (this != &ap)
		{
			if (_ptr)
				delete _ptr;
			_ptr = ap._ptr;
			ap._ptr = nullptr;
		}
		return *this;
	}
	~auto_ptr()
	{
		if (_ptr)
		{
			cout << "delete:" << _ptr;
			delete _ptr;
		}
	}
	T& operator*()
	{
		return *_ptr;

	}
	T* operator->()
	{
		return _ptr;
	}
private:
	T* _ptr;
};



template <class T>
class unique_ptr
{
public:
	unique_ptr(T* ptr)
		:_ptr(ptr)
	{}
	unique_ptr(unique_ptr<T>& ap) = delete;
	unique_ptr<T>& oprator = (unique_ptr<T>&ap) = delete;
	~unique_ptr()
	{
		if (_ptr)
		{
			cout << "delete:" << _ptr;
			delete _ptr;
		}
	}
	T& operator*()
	{
		return *_ptr;

	}
	T* operator->()
	{
		return _ptr;
	}
private:
	T* _ptr;
};

namespace qt
{
	template <class T>
	class shared_ptr
	{
	public:
		shared_ptr(T* ptr)
			:_ptr(ptr)
			, _pcount(new int(1))
		{}
		shared_ptr(shared_ptr<T>& sp)
			:_ptr(sp._ptr)
			, _pcount(sp._pcount)
		{
			++(*_pcount);
		}
	
		~shared_ptr()
		{
			if (--(*pcount) == 0 && _ptr)
			{
				cout << "delete:" << _ptr;
				delete _ptr;
				_ptr = nullptr;
				delete _pcount;
				_pcount = nullptr;
			}
		}
		shared_ptr<T>& operator=(shared_ptr<T>& sp)
		{
			if (this != &sp)
			{
				if (--(*_pcount) == 0)
				{
					delete _pcount;
					delete _ptr;

				}
				_ptr = sp._ptr;
				_pcount = sp.pcount;
				++(*_pcount);
			}
			return *this;
		}
		T& operator*()
		{
			return *_ptr;

		}
		T* operator->()
		{
			return _ptr;
		}
	private:
		T* _ptr;
		int* pcount;
	};
}
```
